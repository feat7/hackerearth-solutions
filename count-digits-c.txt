#include <stdio.h>

int main()
{

	int slen = strlen(str);
	
	for(int i=0; i < slen; i++) {
		switch(str[i]) {
			case '0':
			n0++;
			break;
			case '1':
			n1++;
			break;
			case '2':
			n2++;
			break;
			case '3':
			n3++;
			break;
			case '4':
			n4++;
			break;
			case '5':
			n5++;
			break;
			case '6':
			n6++;
			break;
			case '7':
			n7++;
			break;
			case '8':
			n8++;
			break;
			case '9':
			n9++;
			break;
		}
	}
	
	printf("0 %d\n1 %d\n2 %d\n3 %d\n4 %d\n5 %d\n6 %d\n7 %d\n8 %d\n9 %d", n0, n1, n2, n3, n4, n5, n6, n7, n8, n9);
    
    return 0;
}
